## 6. Дизайн-токены из Figma (подтверждённые)

```css
/* Цвета */
--brand:       #243888;   /* Unbounded заголовки, кнопки, тёмный */
--brand-mid:   #3f559e;   /* Hero background, checkbox border */
--brand-soft:  #6476b6;   /* Badge/sticker background */
--brand-fog:   #6678b6;   /* Final section bg, search button */
--surface:     #f2f3f6;   /* Alternate card backgrounds */
--ink:         #000000;   /* Основной текст */
--muted:       #5c6470;   /* Placeholder text */
--line:        #e0e2ea;   /* Search bar border */
--white:       #ffffff;

/* Типографика Desktop */
--heading-hero:    Unbounded Bold 60px / 1.1
--heading-section: Unbounded Bold 24px / 1.2
--button-text:     Unbounded Bold 14px / 1.5
--body:            Inter Regular 14px / 1.5
--badge:           Inter Regular 20.364px / 1  (opacity 0.5)
--checkbox-label:  Inter Regular 12px

/* Типографика Mobile */
--heading-hero-mobile: Unbounded Bold 40px / 1.1
--badge-mobile:        Inter Regular 14px / 1  (opacity 0.5)
--handwritten:         Caveat Bold 22px / 1.2  (CTA box label)

/* Spacing */
--section-padding-desktop: 60px 0
--section-padding-mobile:  48px 0  (иногда 48px top + 60px bottom в final)
--card-padding:            60px 16px (desktop), 48px 16px (mobile)
--content-gap:             16px (основной gap между элементами)

/* Border radius */
--radius-xl:   32px  (image cards, CTA box)
--radius-pill: 56px  (buttons)
--radius-card: 40px  (search bar)
--radius-badge-desktop: 5.818px
--radius-badge-mobile:  4px
```
